package com.example.callum.first; 
 
import android.content.Intent; 
import android.os.Bundle; 
import android.support.v7.app.AppCompatActivity; 
import android.util.Log; 
import android.view.Menu; 
import android.view.MenuItem; 
import android.view.View; 
import android.widget.Button; 
import android.widget.TextView; 
 
import com.example.callum.first.util.Globalclass; 
 
public class MainActivity extends AppCompatActivity implements View.OnClickListener{ 
 
    private Globalclass g1; 
 
    { 
        g1 = new Globalclass(); 
    } 
 
 
    String image; 
    Button play; 
    TextView level; 
    Button login; 
    Boolean loggedIn; 
 
    @Override 
    protected void onCreate(Bundle savedInstanceState) { 
        super.onCreate(savedInstanceState); 
        setContentView(R.layout.activity_main); 
        play = (Button) findViewById(R.id.Play_button); 
        play.setOnClickListener(this); 
        level=(TextView)findViewById(R.id.level); 
        level.setText("Level: "+g1.getLevel()); 
        login=(Button) findViewById(R.id.Login_Button); 
        login.setOnClickListener(this); 
        loggedIn = true; 
        Log.w("great","message"); 
        if (loggedIn){ 
            login.setHighlightColor(getResources().getColor(R.color.Green)); 
        } 
 
    } 
 
 
 
    @Override 
    public boolean onCreateOptionsMenu(Menu menu) { 
        // Inflate the menu; this adds items to the action bar if it is present. 
        getMenuInflater().inflate(R.menu.menu_main, menu); 
 
        return true; 
    } 
    //hello i am leo and i like this laptop 
    @Override 
    public boolean onOptionsItemSelected(MenuItem item) { 
        // Handle action bar item clicks here. The action bar will 
        // automatically handle clicks on the Home/Up button, so long 
        // as you specify a parent activity in AndroidManifest.xml. 
        int id = item.getItemId(); 
 
        //noinspection SimplifiableIfStatement 
        if (id == R.id.action_settings) { 
            level.setText("boo"); 
 
            return true; 
 
        } 
 
        return super.onOptionsItemSelected(item); 
    } 
 
    private void PlayClick() { 
        if(!loggedIn){ 
            play.setTextColor(getResources().getColor(R.color.Red)); 
            play.setText("Please login first"); 
 
        }else { 
            startActivity(new Intent("android.intent.action.PLAY")); 
        } 
    } 
 
    private void LoginClick(){ 
        play.setTextColor(getResources().getColor(R.color.Black)); 
        play.setText("Play"); 
 
    } 
//leo wilson is the best but we dont have bus money...... 
 
 
    @Override 
    public void onClick(View v) { 
        switch (v.getId()) { 
 
            case R.id.Play_button: 
                PlayClick(); 
                break; 
            case R.id.Login_Button: 
                LoginClick(); 
                break; 
 
        } 
 
    } 
 
    public void newimage() { 
        image = "hello"; 
    } 
 
}
